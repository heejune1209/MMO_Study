## 수치와 문자열

```sql
USE BaseballData;

-- 나이 계산
-- 2021 - 1934
-- 사칙연산 + - * / % 
-- 사칙연산 우선 순위는 동일

-- NULL 타입과 연산은 실행은 되지만 어떤 연산을 하더라도 NULL이 나온다.
SELECT 2021 - NULL;

SELECT 3 / 2; -- 1.5 => 1
SELECT 3.0 / 2 -- 1.5 => 기존 C#과 동일
SELECT 3 / 0 -- error
SELECT ROUND(3.1142324, 3); -- 반올림, 3.1140000
SELECT POWER(2, 3) -- 제곱, 8

SELECT 2021 - birthYear AS koreanAge
FROM players
WHERE deathYear IS NULL AND birthYear IS NOT NULL AND (2021 - birthYear) <= 80
ORDER BY koreanAge;

-- 실행 순서
-- FROM
-- WHERE
-- SELECT
-- ORDERBY
-- 따라서 WHERE에서 koreanAge을 사용 할 수가 없다.

-- 문자열 연산
SELECT 'Hello World'
SELECT '안녕하세요' -- ??

-- 문자 하나당 1바이트이기 때문에 유니코드를 활용하고 싶으면 앞에 N을 붙여줘야 한다.
SELECT N'안녕하세요' -- 정상 출력

-- 문자열을 합치는 연산
-- SQL : CONCAT
-- T-SQL : +
SELECT 'Hello ' + 'World';

-- 1번째에서 4자리 추출
SELECT SUBSTRING('20200425', 1, 4); -- 2020

SELECT TRIM('      Hello World'); -- Hello World

-- 필요할 때마다 그때 그때 찾아보면서 활용해보자

-- 응용
-- 이름 2개를 합쳐서 출력
SELECT nameFirst + ' ' + nameLast AS fullName
FROM players
WHERE nameFirst IS NOT NULL AND nameLast IS NOT NULL;
```
